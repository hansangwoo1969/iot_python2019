 ex) where gpa>=3.0 and class ='3';
  where home = '부산' or home = '서울'

 ex) where gpa>=3.5 and class <>'3';

 ex) where gpa between 2.9 and 3.3
* 검색 조건 리스트 IN, NOT IN
where home IN('부산','광주');
=> where home='부산' or home='광주'
where home NOT IN('부산','광주');
=> where home <> '부산' and home <> '광주'
